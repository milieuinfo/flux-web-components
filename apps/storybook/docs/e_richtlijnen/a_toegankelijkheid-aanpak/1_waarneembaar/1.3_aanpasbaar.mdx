<FluxWcagExample

    title="Gebruik het 'role' attribuut met ARIA-landmarks (region, banner, navigation, ... ) om paginaonderdelen te identificeren."

    <div role="main">the page's main content here</div>

        </div>
        <div>
            <label for="exp-date">Expiry Date:</label>
            <input autocomplete="cc-exp" id="exp-date" type="month">
        </div>
        <div>

        </div>
    </form>
